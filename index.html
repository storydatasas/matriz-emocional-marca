            // Show a specific question

            function showQuestion(index) {

                currentQuestionIndex = index;

                const question = questions[index];

                

                currentQuestionElement.textContent = index + 1;

                progressBar.style.width = `${((index + 1) / questions.length) * 100}%`;

                

                questionsContainer.innerHTML = '';

                

                const questionElement = document.createElement('div');

                questionElement.classList.add('mb-6');

                

                const questionText = document.createElement('h3');

                questionText.classList.add('text-xl', 'font-bold', 'mb-4', 'text-indigo-700');

                questionText.textContent = question.text;

                questionElement.appendChild(questionText);

                

                const optionsContainer = document.createElement('div');

                optionsContainer.classList.add('flex', 'flex-col', 'space-y-3', 'mt-4');

                

                question.options.forEach((option, optionIndex) => {

                    // Check if this option was previously selected

                    if (answers[question.id] === option.value) {

                    }

                    

                    optionElement.innerHTML = `

                // Check if current question is answered

                    showQuestion(currentQuestionIndex + 1);

                } else {

                    showResults();

                }

            }

            // Calculate scores and show results

            function showResults() {

                questionsSection.classList.add('hidden');

                resultsSection.classList.remove('hidden');

                

                // Calculate scores

                let intensityScore = 0;

                let emotionScore = 0;

                let intensityCount = 0;

                let emotionCount = 0;

                

                questions.forEach(question => {

                    if (question.type === 'intensity' && answers[question.id]) {

                        intensityScore += answers[question.id];

                        intensityCount++;

                    } else if (question.type === 'emotion' && answers[question.id]) {

                        emotionScore += answers[question.id];

                        emotionCount++;

                    }

                });

                

                // Normalize scores to 0-100 range

                const normalizedIntensity = intensityCount > 0 ? ((intensityScore / (intensityCount * 5)) * 100) : 50;

                const normalizedEmotion = emotionCount > 0 ? ((emotionScore / (emotionCount * 5)) * 100) : 50;

                

                // Position the brand marker

                const xPos = normalizedIntensity / 100 * 500; // 500px is the width of the matrix

                const yPos = (100 - normalizedEmotion) / 100 * 500; // Invert Y axis

                

                brandMarker.style.left = `${xPos}px`;

                brandMarker.style.top = `${yPos}px`;

                

                // Determine quadrant

                let quadrant;

                if (normalizedIntensity >= 50 && normalizedEmotion < 50) {

                    quadrant = 1; // High Pain - High Intensity

                } else if (normalizedIntensity >= 50 && normalizedEmotion >= 50) {

                    quadrant = 2; // High Pleasure - High Intensity

                } else if (normalizedIntensity < 50 && normalizedEmotion < 50) {

                    quadrant = 3; // Soft Pain - Low Intensity

                } else {

                    quadrant = 4; // Soft Pleasure - Low Intensity

                }

                

                // Update result elements

                const quadrantData = quadrants[quadrant];

                

                document.getElementById('quadrantResult').textContent = quadrantData.title;

                document.getElementById('quadrantResult').className = quadrantData.bgColor + ' p-5 rounded-lg mb-4 text-white';

                

                document.getElementById('narrativeResult').textContent = quadrantData.narrative;

                document.getElementById('keywordsResult').textContent = quadrantData.keywords;

                document.getElementById('strategyResult').textContent = quadrantData.strategy;

                document.getElementById('idealForResult').textContent = quadrantData.idealFor;

                document.getElementById('exampleResult').textContent = quadrantData.example;

                

                const recommendationsElement = document.getElementById('recommendationsResult');

                recommendationsElement.innerHTML = '';

                

                quadrantData.recommendations.forEach(recommendation => {

                    const li = document.createElement('li');

                    li.textContent = recommendation;

                    recommendationsElement.appendChild(li);

                });

                

                // Scroll to results

                resultsSection.scrollIntoView({ behavior: 'smooth' });

            }

        });
